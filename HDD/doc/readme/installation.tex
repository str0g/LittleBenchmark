\subsubsection{From Repository(not uploaded yet)}

\begin{verbatim}
sudo add-apt-repository ppa:buskol-waw-pl/littlebenchmark-hdd
sudo apt-get update
sudo apt-get install littlebenchmark-hdd
\end{verbatim}
